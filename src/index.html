gbTurntable.init({
    id: 'turntable',
    config: function(callback){
        // 获取奖品信息
        // 奖项 text 属性不能为空，用于显示或抽中奖品提示
        // img 为奖品图片地址，如果不为空则转盘奖品按图片方式显示
         callback && callback([{ text: '1 premio', img: 'images/redpacket.png' }, { text: '2 premio' }, { text: '3 premio' }, { text: '显示器',
        img: 'images/display.png' }, { text: '5 premio' }, { text: '6 premio' }])
    },
    getPrize: function(callback) {
        // 获取中奖信息
        var num = Math.random() * 6 >>> 0,   //奖品ID
            chances = num;  // 可抽奖次数
            callback && callback([num, chances]);
    },
    gotBack: function(data) {
        alert('恭喜抽中' + data);
    }
});
